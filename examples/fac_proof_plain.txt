// 1 = fac(0) and 0 <= n
p = 1;
// p = fac(0) and 1 <= n + 1
c = 1;
// p = fac(c-1) and c <= n + 1
while(c <= n){
    //inv: p = fac(c-1) and c <= n + 1
    // p = fac(c-1) and c <= n + 1 and c <= n
    // p = fac(c-1) and c <= n
    // weakening
    // p * c = fac(c) and c+1 <= n+1
    p = p * c;
    // p = fac(c) and c+1 <= n+1
    c = c + 1;
    // p = fac(c-1) and c <= n+1 
}
// p = fac(c-1) and c <= n+1 and c > n
// p = fac(c-1) and c = n+1
